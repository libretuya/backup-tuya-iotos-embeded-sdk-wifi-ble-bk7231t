Typedef mcast_fixed_head_t;


      
typedef struct {
   uint8_t version: 4;
   uint8_t reserved: 4;
   uint8_t reserved2;
   uint32_t sequence;
   uint32_t type;
   uint32_t id;//主控或者受控设备自己的id
   uint32_t total_len;//报文总的长度,从head到tail
   uint32_t frame_num;//frame的个数
} mcast_fixed_head_t;
